### 📝 **NLP Movie Review Analyzer** 📝
```python
# 🎬 Advanced Text Classification with NLTK
import nltk
from sklearn.feature_extraction.text import TfidfVectorizer
from sklearn.naive_bayes import MultinomialNB

class MovieReviewClassifier:
    def __init__(self):
        self.vectorizer = TfidfVectorizer(max_features=5000)
        self.classifier = MultinomialNB()
        self.processed_reviews = 0
        
    def preprocess_text(self, text):
        # Advanced preprocessing pipeline
        tokens = nltk.word_tokenize(text.lower())
        tokens = [word for word in tokens if word.isalpha()]
        return ' '.join(tokens)
    
    def analyze_sentiment(self, review):
        processed = self.preprocess_text(review)
        sentiment = self.classifier.predict([processed])[0]
        confidence = max(self.classifier.predict_proba([processed])[0])
        
        self.processed_reviews += 1
        return f"📊 Sentiment: {sentiment} | Confidence: {confidence:.2f}"
    
    def get_stats(self):
        return f"🎯 Analyzed {self.processed_reviews} movie reviews and counting!"
```
